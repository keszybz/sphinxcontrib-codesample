Show case¶

plain¶

x = 2
y = x**2
print(y)

>>> x = 2
>>> y = x**2
>>> print(y)
4

multiline string¶

s = '''
a b c
d e f
'''
print(s)

>>> s = '''
... a b c
... d e f
... '''
>>> print(s)

a b c
d e f

multiline function¶

def f(x, y, *, z=11):
    "Return the frobnication of x, y, and z"
    return x + y + z
y = f(1, 2, z=3)
print(y)

>>> def f(x, y, *, z=11):
...     "Return the frobnication of x, y, and z"
...     return x + y + z
>>> y = f(1, 2, z=3)
>>> print(y)
6

suppress inline¶

def ellipsized_function(x, y, *, z=11):
    "Return the frobnication of x, y, and z" # SUPPRESS
    ...
    return x + y + z                         # SUPPRESS
y = ellipsized_function(1, 2, z=3)
print(y)

>>> def ellipsized_function(x, y, *, z=11):
>>> y = ellipsized_function(1, 2, z=3)
>>> print(y)
6

suppress block¶

:suppress:

x = 1
y = 2
print('this should be the only line visible')

this should be the only line visible

unsuppress block¶

:nosuppress:

x = 1
y = 2
print('this should be the third line of code')

>>> x = 1
>>> y = 2
>>> print('this should be the third line of code')
this should be the third line of code
